## ðŸ“ SAMPLE IMPLEMENTATION: FILE UPLOAD COMPONENT
================================================================================

Here's a starter implementation for the file upload:

```jsx
// src/components/FileUpload.jsx
import React, { useState, useRef } from 'react';
import { Upload, File, X, CheckCircle } from 'lucide-react';

const ACCEPTED_TYPES = {
  'application/pdf': 'PDF',
  'text/plain': 'TXT',
  'image/png': 'PNG',
  'image/jpeg': 'JPEG',
  'application/vnd.openxmlformats-officedocument.wordprocessingml.document': 'DOCX'
};

const FileUpload = ({ onFilesProcessed }) => {
  const [isDragging, setIsDragging] = useState(false);
  const [files, setFiles] = useState([]);
  const [processing, setProcessing] = useState(false);
  const inputRef = useRef(null);

  const handleDrop = async (e) => {
    e.preventDefault();
    setIsDragging(false);
    
    const droppedFiles = Array.from(e.dataTransfer.files);
    processFiles(droppedFiles);
  };

  const processFiles = async (fileList) => {
    setProcessing(true);
    const processed = [];
    
    for (const file of fileList) {
      if (ACCEPTED_TYPES[file.type]) {
        // Extract content based on file type
        const content = await extractContent(file);
        processed.push({
          id: crypto.randomUUID(),
          name: file.name,
          type: ACCEPTED_TYPES[file.type],
          size: file.size,
          content,
          uploadDate: new Date().toISOString()
        });
      }
    }
    
    setFiles(prev => [...prev, ...processed]);
    setProcessing(false);
    onFilesProcessed?.(processed);
  };

  const extractContent = async (file) => {
    // For text files
    if (file.type === 'text/plain') {
      return await file.text();
    }
    
    // For PDFs - would use pdfjs-dist
    // For images - would use tesseract.js
    // Placeholder for now
    return `[Content of ${file.name}]`;
  };

  return (
    <div
      onDragOver={(e) => { e.preventDefault(); setIsDragging(true); }}
      onDragLeave={() => setIsDragging(false)}
      onDrop={handleDrop}
      onClick={() => inputRef.current?.click()}
      className={`
        border-2 border-dashed rounded-2xl p-8 text-center cursor-pointer
        transition-all duration-300
        ${isDragging 
          ? 'border-accent bg-accent/5 scale-[1.02]' 
          : 'border-black/10 hover:border-black/20'}
      `}
    >
      <input
        ref={inputRef}
        type="file"
        multiple
        accept={Object.keys(ACCEPTED_TYPES).join(',')}
        onChange={(e) => processFiles(Array.from(e.target.files))}
        className="hidden"
      />
      
      <Upload size={32} className="mx-auto mb-4 text-secondary" />
      <p className="font-medium text-primary">Drop files here or click to upload</p>
      <p className="text-sm text-secondary mt-1">
        Supports PDF, DOCX, TXT, PNG, JPEG
      </p>
      
      {files.length > 0 && (
        <div className="mt-4 space-y-2">
          {files.map(file => (
            <div key={file.id} className="flex items-center gap-2 text-sm p-2 bg-white rounded-lg">
              <File size={16} className="text-accent" />
              <span className="flex-1 text-left truncate">{file.name}</span>
              <CheckCircle size={16} className="text-green-500" />
            </div>
          ))}
        </div>
      )}
    </div>
  );
};

export default FileUpload;
```
